16fn get_url_file_name(url: &str) -> PyResult<String> {
17    let url =
18        deviantart::Url::parse(url).map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
19    let file_name = url
20        .path_segments()
21        .ok_or_else(|| PyRuntimeError::new_err("missing path"))?
22        .next_back()
23        .ok_or_else(|| PyRuntimeError::new_err("missing file name"))?;
24
25    Ok(file_name.to_string())
26}

173    /// Get metadata for a deviation.
174    pub fn get_deviation(&self, source: Bound<'_, PyAny>) -> PyResult<Deviation> {
175        let tokio_rt = TOKIO_RT
176            .as_ref()
177            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
178
179        let source = if let Ok(url) = source.cast::<PyString>() {
180            url.to_cow()?.into_owned()
181        } else if let Ok(id) = source.extract::<u64>() {
182            format!("https://www.deviantart.com/view/{id}")
183        } else {
184            return Err(PyValueError::new_err(
185                "source must be a deviation id or a url",
186            ));
187        };
188
189        let scraped_webpage_info = tokio_rt
190            .block_on(async { self.client.scrape_webpage(source.as_str()).await })
191            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
192
193        let current_deviation = scraped_webpage_info
194            .get_current_deviation()
195            .ok_or_else(|| PyRuntimeError::new_err("failed to get current deviation"))?;
196
197        let current_deviation_extended = scraped_webpage_info
198            .get_current_deviation_extended()
199            .ok_or_else(|| PyRuntimeError::new_err("failed to get current deviation extended"))?;
200
201        let download_url = current_deviation_extended
202            .download
203            .as_ref()
204            .map(|download| download.url.clone())
205            .or_else(|| current_deviation.get_download_url())
206            .map(String::from);
207
208        let fullview_url = current_deviation
209            .get_fullview_url(Default::default())
210            .context("failed to get fullview url")
211            .map(String::from)?;
212
213        let additional_media_download_urls = current_deviation_extended
214            .additional_media
215            .as_ref()
216            .map(|additional_media| {
217                additional_media
218                    .iter()
219                    .map(|additional_media| {
220                        if !current_deviation_extended.can_download_additional_media() {
221                            return None;
222                        }
223
224                        additional_media
225                            .media
226                            .base_uri
227                            .clone()
228                            .map(|mut url| {
229                                // Some images require a token, some don't.
230                                // I don't know what causes the token to be required.
231                                // Regardless, always sending a token when possible doesn't seem to cause issues.
232                                match additional_media.media.token.first().as_ref() {
233                                    Some(token) => {
234                                        url.query_pairs_mut().append_pair("token", token);
235                                        url
236                                    }
237                                    None => url,
238                                }
239                            })
240                            .map(String::from)
241                    })
242                    .collect()
243            });
244        let additional_media_fullview_urls = current_deviation_extended
245            .additional_media
246            .as_ref()
247            .map(|additional_media| {
248                additional_media
249                    .iter()
250                    .map(|additional_media| {
251                        additional_media
252                            .media
253                            .get_fullview_url(deviantart::GetFullviewUrlOptions {
254                                strp: Some(false),
255                                quality: Some(100),
256                                png: true,
257                            })
258                            .map(String::from)
259                    })
260                    .collect::<Result<_, _>>()
261            })
262            .transpose()
263            .context("failed to get fullview urls for additional media")?;
264
265        Ok(Deviation {
266            id: current_deviation.deviation_id,
267            title: current_deviation.title.clone(),
268            description: current_deviation_extended.description.clone(),
269            kind: current_deviation.kind.clone(),
270            download_url,
271            fullview_url,
272            additional_media_download_urls,
273            additional_media_fullview_urls,
274        })
275    }

277    /// Download a deviation.
278    #[pyo3(signature = (deviation, use_fullview=false))]
279    pub fn download_deviation<'p>(
280        &self,
281        deviation: &Deviation,
282        use_fullview: bool,
283        py: Python<'p>,
284    ) -> PyResult<Bound<'p, PyBytes>> {
285        let tokio_rt = TOKIO_RT
286            .as_ref()
287            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
288
289        let url = if use_fullview {
290            &deviation.fullview_url
291        } else {
292            deviation
293                .download_url
294                .as_ref()
295                .ok_or(PyValueError::new_err("deviation is missing a download url"))?
296        };
297
298        let bytes = tokio_rt
299            .block_on(async {
300                self.client
301                    .client
302                    .get(url)
303                    .send()
304                    .await?
305                    .error_for_status()?
306                    .bytes()
307                    .await
308            })
309            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
310
311        Ok(PyBytes::new(py, &bytes))
312    }

364    /// Get the folder given by the url.
365    pub fn get_folder(&self, url: &str) -> PyResult<Folder> {
366        let tokio_rt = TOKIO_RT
367            .as_ref()
368            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
369
370        let scraped_webpage_info = tokio_rt
371            .block_on(async { self.client.scrape_webpage(url).await })
372            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
373
374        let id = scraped_webpage_info
375            .get_current_folder_id()
376            .ok_or_else(|| PyRuntimeError::new_err("missing folder id"))?;
377
378        let stream = scraped_webpage_info
379            .get_folder_deviations_stream(id)
380            .ok_or_else(|| PyRuntimeError::new_err("missing folder deviation stream"))?;
381
382        let mut deviation_ids = stream.items.clone();
383
384        let folder_entity = scraped_webpage_info
385            .get_gallery_folder_entity(id)
386            .ok_or_else(|| PyRuntimeError::new_err("missing gallery folder entity"))?;
387
388        let user_entity = scraped_webpage_info
389            .get_user_entity(folder_entity.owner)
390            .ok_or_else(|| PyRuntimeError::new_err("missing user entity"))?;
391
392        let owner_name = user_entity.username.clone();
393
394        if stream.has_more {
395            tokio_rt
396                .block_on(async {
397                    let mut has_more = true;
398                    while has_more {
399                        let offset = u64::try_from(deviation_ids.len()).unwrap();
400                        let response = self
401                            .client
402                            .list_folder_contents(
403                                &owner_name,
404                                id,
405                                offset,
406                                &scraped_webpage_info.config.csrf_token,
407                            )
408                            .await?;
409                        deviation_ids.extend(
410                            response
411                                .results
412                                .iter()
413                                .map(|deviation| deviation.deviation_id),
414                        );
415                        has_more = response.has_more;
416                    }
417
418                    Result::<_, deviantart::Error>::Ok(())
419                })
420                .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
421        }
422
423        Ok(Folder {
424            id,
425            name: folder_entity.name.clone(),
426            owner_name,
427            deviation_ids,
428        })
429    }
