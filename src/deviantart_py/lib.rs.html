15fn get_url_file_name(url: &str) -> PyResult<String> {
16    let url =
17        deviantart::Url::parse(url).map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
18    let file_name = url
19        .path_segments()
20        .ok_or_else(|| PyRuntimeError::new_err("missing path"))?
21        .next_back()
22        .ok_or_else(|| PyRuntimeError::new_err("missing file name"))?;
23
24    Ok(file_name.to_string())
25}

126    /// Get metadata for a deviation.
127    pub fn get_deviation(&self, source: Bound<'_, PyAny>) -> PyResult<Deviation> {
128        let tokio_rt = TOKIO_RT
129            .as_ref()
130            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
131
132        let source = if let Ok(url) = source.downcast::<PyString>() {
133            url.to_cow()?.into_owned()
134        } else if let Ok(id) = source.extract::<u64>() {
135            format!("https://www.deviantart.com/view/{id}")
136        } else {
137            return Err(PyValueError::new_err(
138                "source must be a deviation id or a url",
139            ));
140        };
141
142        let scraped_webpage_info = tokio_rt
143            .block_on(async { self.client.scrape_webpage(source.as_str()).await })
144            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
145
146        let current_deviation = scraped_webpage_info
147            .get_current_deviation()
148            .ok_or_else(|| PyRuntimeError::new_err("failed to get current deviation"))?;
149
150        let current_deviation_extended = scraped_webpage_info
151            .get_current_deviation_extended()
152            .ok_or_else(|| PyRuntimeError::new_err("failed to get current deviation extended"))?;
153
154        let download_url = current_deviation_extended
155            .download
156            .as_ref()
157            .map(|download| download.url.clone())
158            .or_else(|| current_deviation.get_download_url())
159            .map(String::from);
160
161        let fullview_url = current_deviation.get_fullview_url().map(String::from);
162
163        let additional_media_download_urls = current_deviation_extended
164            .additional_media
165            .as_ref()
166            .map(|additional_media| {
167                additional_media
168                    .iter()
169                    .map(|additional_media| {
170                        if current_deviation_extended.is_da_protected.unwrap_or(false) {
171                            return None;
172                        }
173
174                        additional_media
175                            .media
176                            .base_uri
177                            .clone()
178                            .map(|mut url| {
179                                // Some images require a token, some don't.
180                                // I don't know what causes the token to be required.
181                                // Regardless, always sending a token when possible doesn't seem to cause issues.
182                                match additional_media.media.token.first().as_ref() {
183                                    Some(token) => {
184                                        url.query_pairs_mut().append_pair("token", token);
185                                        url
186                                    }
187                                    None => url,
188                                }
189                            })
190                            .map(String::from)
191                    })
192                    .collect()
193            });
194        let additional_media_fullview_urls = current_deviation_extended
195            .additional_media
196            .as_ref()
197            .map(|additional_media| {
198                additional_media
199                    .iter()
200                    .map(|additional_media| {
201                        additional_media.media.get_fullview_url().map(String::from)
202                    })
203                    .collect()
204            });
205
206        Ok(Deviation {
207            id: current_deviation.deviation_id,
208            title: current_deviation.title.clone(),
209            description: current_deviation_extended.description.clone(),
210            kind: current_deviation.kind.clone(),
211            download_url,
212            fullview_url,
213            additional_media_download_urls,
214            additional_media_fullview_urls,
215        })
216    }

218    /// Download a deviation.
219    #[pyo3(signature = (deviation, use_fullview=false))]
220    pub fn download_deviation<'p>(
221        &self,
222        deviation: &Deviation,
223        use_fullview: bool,
224        py: Python<'p>,
225    ) -> PyResult<Bound<'p, PyBytes>> {
226        let tokio_rt = TOKIO_RT
227            .as_ref()
228            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
229
230        let url = if use_fullview {
231            deviation
232                .fullview_url
233                .as_ref()
234                .ok_or(PyValueError::new_err("deviation is missing a fullview url"))?
235        } else {
236            deviation
237                .download_url
238                .as_ref()
239                .ok_or(PyValueError::new_err("deviation is missing a download url"))?
240        };
241
242        let bytes = tokio_rt
243            .block_on(async {
244                self.client
245                    .client
246                    .get(url)
247                    .send()
248                    .await?
249                    .error_for_status()?
250                    .bytes()
251                    .await
252            })
253            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
254
255        Ok(PyBytes::new(py, &bytes))
256    }

308    /// Get the folder given by the url.
309    pub fn get_folder(&self, url: &str) -> PyResult<Folder> {
310        let tokio_rt = TOKIO_RT
311            .as_ref()
312            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
313
314        let scraped_webpage_info = tokio_rt
315            .block_on(async { self.client.scrape_webpage(url).await })
316            .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
317
318        let id = scraped_webpage_info
319            .get_current_folder_id()
320            .ok_or_else(|| PyRuntimeError::new_err("missing folder id"))?;
321
322        let stream = scraped_webpage_info
323            .get_folder_deviations_stream(id)
324            .ok_or_else(|| PyRuntimeError::new_err("missing folder deviation stream"))?;
325
326        let mut deviation_ids = stream.items.clone();
327
328        let folder_entity = scraped_webpage_info
329            .get_gallery_folder_entity(id)
330            .ok_or_else(|| PyRuntimeError::new_err("missing gallery folder entity"))?;
331
332        let user_entity = scraped_webpage_info
333            .get_user_entity(folder_entity.owner)
334            .ok_or_else(|| PyRuntimeError::new_err("missing user entity"))?;
335
336        let owner_name = user_entity.username.clone();
337
338        if stream.has_more {
339            tokio_rt
340                .block_on(async {
341                    let mut has_more = true;
342                    while has_more {
343                        let offset = u64::try_from(deviation_ids.len()).unwrap();
344                        let response = self
345                            .client
346                            .list_folder_contents(
347                                &owner_name,
348                                id,
349                                offset,
350                                &scraped_webpage_info.config.csrf_token,
351                            )
352                            .await?;
353                        deviation_ids.extend(
354                            response
355                                .results
356                                .iter()
357                                .map(|deviation| deviation.deviation_id),
358                        );
359                        has_more = response.has_more;
360                    }
361
362                    Result::<_, deviantart::Error>::Ok(())
363                })
364                .map_err(|error| PyRuntimeError::new_err(error.to_string()))?;
365        }
366
367        Ok(Folder {
368            id,
369            name: folder_entity.name.clone(),
370            owner_name,
371            deviation_ids,
372        })
373    }
